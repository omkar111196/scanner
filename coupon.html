// Initialize Firebase firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); // Get coupon ID from URL const urlParams = new URLSearchParams(window.location.search); const couponId = urlParams.get('id'); // Check coupon status db.collection("coupons").doc(couponId).get().then((doc) => { if (doc.exists) { const coupon = doc.data(); if (coupon.status === 'used') { document.getElementById("coupon-status").innerHTML = `

This coupon has already been used.

`; } else { document.getElementById("coupon-status").innerHTML = `

Coupon is valid.

`; document.getElementById("verify-section").style.display = "block"; } } else { document.getElementById("coupon-status").innerHTML = `

Invalid coupon ID.

`; } }).catch((error) => { document.getElementById("coupon-status").innerHTML = `

Error retrieving coupon: ${error.message}

`; }); // Employee verification function verifyEmployee() { const password = document.getElementById("employee-password").value; if (password === 'omkar1234') { document.getElementById("user-details").style.display = "block"; } else { alert('Invalid employee password'); } } // Submit user details and mark coupon as used function submitDetails() { const name = document.getElementById("name").value; const gender = document.getElementById("gender").value; const phone = document.getElementById("phone").value; if (!name || !gender || !phone) { alert("Please fill in all user details"); return; } // Update Firestore with user details and mark coupon as used db.collection("coupons").doc(couponId).update({ status: 'used', verified: true, user_details: { name: name, gender: gender, phone: phone } }).then(() => { alert('Coupon successfully verified and used!'); document.getElementById("coupon-status").innerHTML = `

Coupon has been used successfully.

`; document.getElementById("verify-section").style.display = "none"; document.getElementById("user-details").style.display = "none"; }).catch((error) => { console.error("Error updating coupon: ", error); }); }
